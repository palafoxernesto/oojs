const COINS = [200, 100, 50, 20, 10, 5, 2, 1]
        
var getChange = function (totalPayable, cashPaid) {
   difference = cashPaid - totalPayable;
   var change = []
 
   COINS.forEach(function(coin) {
       while (difference >= coin){
           change.push(coin);
           difference = difference - coin;
       }
   });
    return change;
}
